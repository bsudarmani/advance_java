import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class emi extends JFrame implements ActionListener {

    JTextField txtAmount, txtRate, txtMonths, txtEmi;
    JButton btnCalculate;

    public emi() {

        setTitle("EMI Calculator");
        setSize(380, 300);
        setLayout(null);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLocationRelativeTo(null);

        JLabel l1 = new JLabel("Loan Amount:");
        JLabel l2 = new JLabel("Monthly Interest (%):");
        JLabel l3 = new JLabel("Number of Months:");
        JLabel l4 = new JLabel("Monthly EMI:");

        l1.setBounds(30, 30, 160, 25);
        l2.setBounds(30, 70, 160, 25);
        l3.setBounds(30, 110, 160, 25);
        l4.setBounds(30, 200, 160, 25);

        add(l1); add(l2); add(l3); add(l4);

        txtAmount = new JTextField();
        txtRate = new JTextField();
        txtMonths = new JTextField();
        txtEmi = new JTextField();

        txtEmi.setEditable(false);

        txtAmount.setBounds(200, 30, 130, 25);
        txtRate.setBounds(200, 70, 130, 25);
        txtMonths.setBounds(200, 110, 130, 25);
        txtEmi.setBounds(200, 200, 130, 25);

        add(txtAmount);
        add(txtRate);
        add(txtMonths);
        add(txtEmi);

        btnCalculate = new JButton("Calculate EMI");
        btnCalculate.setBounds(110, 150, 150, 30);
        add(btnCalculate);

        btnCalculate.addActionListener(this);

        setVisible(true);
    }

    public void actionPerformed(ActionEvent e) {

        double principal = Double.parseDouble(txtAmount.getText());
        double rate = Double.parseDouble(txtRate.getText()) / 100;
        int months = Integer.parseInt(txtMonths.getText());

        double emi = (principal * rate * Math.pow(1 + rate, months)) /
                (Math.pow(1 + rate, months) - 1);

        txtEmi.setText(String.format("%.2f", emi));
    }

    public static void main(String[] args) {
        new emi();
    }
}

import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class EmojiRadioApp extends JFrame implements ActionListener {

    JRadioButton happyBtn, sadBtn, angryBtn, surprisedBtn;
    JLabel imageLabel;

    public EmojiRadioApp() {

        // Frame settings
        setTitle("Emoji Selector Application");
        setSize(600, 400);
        setLayout(new BorderLayout());
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        // Left Panel (Radio Buttons)
        JPanel leftPanel = new JPanel();
        leftPanel.setLayout(new GridLayout(4, 1));
        leftPanel.setBorder(BorderFactory.createTitledBorder("Select Emotion"));

        happyBtn = new JRadioButton("Happy");
        sadBtn = new JRadioButton("Sad");
        angryBtn = new JRadioButton("Angry");
        surprisedBtn = new JRadioButton("Surprised");

        ButtonGroup bg = new ButtonGroup();
        bg.add(happyBtn);
        bg.add(sadBtn);
        bg.add(angryBtn);
        bg.add(surprisedBtn);

        leftPanel.add(happyBtn);
        leftPanel.add(sadBtn);
        leftPanel.add(angryBtn);
        leftPanel.add(surprisedBtn);

        // Right Panel (Image Display)
        JPanel rightPanel = new JPanel();
        rightPanel.setBorder(BorderFactory.createTitledBorder("Emoji"));
        imageLabel = new JLabel();
        rightPanel.add(imageLabel);

        // Add panels to frame
        add(leftPanel, BorderLayout.WEST);
        add(rightPanel, BorderLayout.CENTER);

        // Add listeners
        happyBtn.addActionListener(this);
        sadBtn.addActionListener(this);
        angryBtn.addActionListener(this);
        surprisedBtn.addActionListener(this);

        setVisible(true);
    }

    @Override
    public void actionPerformed(ActionEvent e) {

        if (happyBtn.isSelected()) {
            imageLabel.setIcon(new ImageIcon(getClass().getResource("/images/happy.jpg")));
        }
        else if (sadBtn.isSelected()) {
            imageLabel.setIcon(new ImageIcon(getClass().getResource("/images/sad.jpg")));
        }
        else if (angryBtn.isSelected()) {
            imageLabel.setIcon(new ImageIcon(getClass().getResource("/images/angry.png")));
        }
        else if (surprisedBtn.isSelected()) {
            imageLabel.setIcon(new ImageIcon(getClass().getResource("/images/surprised.jpg")));
        }

    }

    public static void main(String[] args) {
        new EmojiRadioApp();
    }
}

import javax.swing.*;
import java.awt.*;
public class ProductList extends JFrame {
    JList<String> pro;
    JScrollPane scr;
    JLabel l1, l2, l3, l4;
    ProductList() {
        setSize(500, 250);
        setTitle("ProdctList");
        setResizable(false);
        setLayout(new GridLayout(4, 3, 5, 5));
        l1 = new JLabel("Product Price of ", SwingConstants.CENTER);
        l2 = new JLabel("Select the Product : ");
        l3 = new JLabel("Price : ");
        l4 = new JLabel();
        String[] product = {"Pen", "Pencil", "Scale", "Eraser", "Note", "Paper","A4sheetPaperBundle","ClgBag"};
        pro = new JList<>(product);
        scr = new JScrollPane(pro);
        add(new JLabel(""));
        add(l1);
        add(new JLabel(""));
        add(l2);
        add(new JLabel(""));
        add(scr);
        add(l3);
        add(new JLabel(""));
        add(l4);
        add(new JLabel(""));
        pro.addListSelectionListener(e -> {
            if (!e.getValueIsAdjusting()) {
                Price(pro.getSelectedValue());
            }
        });
        setVisible(true);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    }
    void Price(String product) {
        if (product == null) return;
        switch (product) {
            case "Pen" -> l4.setText("₹ 10");
            case "Scale","Pencil" -> l4.setText("₹ 5");
            case "Eraser" -> l4.setText("₹ 5");
            case "Note" -> l4.setText("₹ 40");
            case "Paper" -> l4.setText("₹ 20");
            case "A4sheetPaperBundle"->l4.setText("₹40");
            case "ClgBag"->l4.setText("₹900");
        }
    }
    public static void main(String[] args) {
        new ProductList();
    }
}
