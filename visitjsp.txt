<%
    // Get visitor count from application scope
    Integer count = (Integer) application.getAttribute("visitorCount");

    // Initialize if null
    if (count == null) {
        count = 0;
    }

    // Count only new session visitors
    if (session.isNew()) {
        count++;
        application.setAttribute("visitorCount", count);
    }
%>
